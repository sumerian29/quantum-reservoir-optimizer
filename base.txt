🧩 Detailed Module Design & Code
1. qubo_builder.py – Convert reservoir physics to QUBO
python
import numpy as np
from typing import Tuple

class QUBOBuilder:
    """
    Constructs QUBO matrix Q from well production potentials P_i
    and interference coefficients I_ij.
    Objective: Maximize sum(P_i x_i) - sum(I_ij x_i x_j)
    Equivalent to minimize: -sum(P_i x_i) + sum(I_ij x_i x_j)
    """
    def __init__(self, production: np.ndarray, interference: np.ndarray):
        """
        production: shape (n,) – production potential for each candidate location
        interference: shape (n, n) – interference penalty (symmetric, zero diagonal)
        """
        self.n = len(production)
        self.P = production
        self.I = interference
        assert self.I.shape == (self.n, self.n), "Interference matrix shape mismatch"
    
    def build_qubo(self) -> np.ndarray:
        """Returns Q matrix (n x n) for QUBO: x^T Q x"""
        Q = np.zeros((self.n, self.n))
        # Diagonal: -P_i  (since we minimize -P_i x_i)
        for i in range(self.n):
            Q[i, i] = -self.P[i]
        # Off-diagonal: I_ij (since we minimize +I_ij x_i x_j)
        for i in range(self.n):
            for j in range(i+1, self.n):
                Q[i, j] = self.I[i, j]
                Q[j, i] = self.I[j, i]   # keep symmetric
        return Q
    
    @staticmethod
    def from_reservoir_properties(porosity: np.ndarray, 
                                   permeability: np.ndarray,
                                   pressure: np.ndarray,
                                   distance_matrix: np.ndarray,
                                   alpha: float = 0.7, beta: float = 0.3) -> Tuple[np.ndarray, np.ndarray]:
        """
        Derive P_i and I_ij from physical properties.
        P_i = alpha*porosity_i + beta*permeability_i + (1-alpha-beta)*pressure_i
        I_ij = 1 / (1 + distance_ij)  (inverse distance penalty)
        """
        # Normalize each property to [0,1]
        por_norm = (porosity - porosity.min()) / (porosity.max() - porosity.min() + 1e-8)
        perm_norm = (permeability - permeability.min()) / (permeability.max() - permeability.min() + 1e-8)
        pres_norm = (pressure - pressure.min()) / (pressure.max() - pressure.min() + 1e-8)
        production = alpha * por_norm + beta * perm_norm + (1 - alpha - beta) * pres_norm
        interference = 1.0 / (1.0 + distance_matrix)
        np.fill_diagonal(interference, 0.0)  # no self-interference
        return production, interference
2. quantum_qaoa/qaoa_circuit.py – QAOA implementation (Qiskit)
python
from qiskit import QuantumCircuit
from qiskit.circuit import Parameter
from qiskit.quantum_info import SparsePauliOp
import numpy as np

def build_qaoa_circuit(qubo_matrix: np.ndarray, gamma: float, beta: float, p: int) -> QuantumCircuit:
    """
    Creates a QAOA circuit for the given QUBO.
    QUBO is first mapped to an Ising Hamiltonian: H = sum_i h_i Z_i + sum_{i<j} J_ij Z_i Z_j.
    """
    n = qubo_matrix.shape[0]
    # Convert QUBO to Ising: x = (1 - Z)/2
    # x^T Q x = constant + sum_i (Q_ii/2 + sum_j Q_ij/4) Z_i + sum_{i<j} (Q_ij/4) Z_i Z_j
    h = np.zeros(n)
    J = np.zeros((n, n))
    const = 0.0
    for i in range(n):
        h[i] = qubo_matrix[i, i] / 2.0
        const += qubo_matrix[i, i] / 4.0
        for j in range(i+1, n):
            J[i, j] = qubo_matrix[i, j] / 4.0
            J[j, i] = J[i, j]
            const += qubo_matrix[i, j] / 4.0
    # Build Hamiltonian as Pauli strings
    pauli_list = []
    coeffs = []
    for i in range(n):
        pauli_list.append(('Z'*i + 'Z' + 'I'*(n-i-1), h[i]))
        coeffs.append(h[i])
    for i in range(n):
        for j in range(i+1, n):
            pauli_str = ['I']*n
            pauli_str[i] = 'Z'
            pauli_str[j] = 'Z'
            pauli_list.append((''.join(pauli_str), J[i,j]))
            coeffs.append(J[i,j])
    cost_ham = SparsePauliOp.from_list(pauli_list)
    
    # QAOA circuit
    qc = QuantumCircuit(n, n)
    # Initial superposition
    qc.h(range(n))
    # p layers
    for _ in range(p):
        # Cost layer: exp(-i gamma H_C)
        qc.append(cost_ham.to_circuit(gamma), range(n))
        # Mixer layer: exp(-i beta sum X_i)
        for i in range(n):
            qc.rx(2*beta, i)
    qc.measure_all()
    return qc, const
3. optimization.py – Hybrid orchestrator
python
from qubo_builder import QUBOBuilder
from classical_model.simulated_annealing import simulated_annealing
from quantum_qaoa.qaoa_optimizer import run_qaoa
import numpy as np

class HybridOptimizer:
    def __init__(self, production, interference, config):
        self.qubo_builder = QUBOBuilder(production, interference)
        self.Q = self.qubo_builder.build_qubo()
        self.config = config
    
    def solve_classical(self, method='sa'):
        if method == 'sa':
            return simulated_annealing(self.Q, max_iter=1000)
        elif method == 'ga':
            from classical_model.genetic_algorithm import genetic_algorithm
            return genetic_algorithm(self.Q)
    
    def solve_quantum(self, p_layers=2, backend='qasm_simulator'):
        return run_qaoa(self.Q, p=p_layers, backend=backend)
    
    def compare_and_report(self):
        classical_solution = self.solve_classical()
        quantum_solution = self.solve_quantum(p_layers=self.config.get('p', 2))
        classical_obj = classical_solution['energy']
        quantum_obj = quantum_solution['energy']
        # Objective is negative of original max (since QUBO minimizes)
        improvement = (classical_obj - quantum_obj) / abs(classical_obj) * 100
        return {
            'classical': classical_solution,
            'quantum': quantum_solution,
            'improvement_%': improvement,
            'best_wells': quantum_solution['x']
        }
4. intelligent_selector.py – ML‑based solver recommendation
python
from sklearn.ensemble import RandomForestClassifier
import numpy as np

class IntelligentSolverSelector:
    """
    Trained on problem features (n, density, condition number) to recommend
    SA, GA, or QAOA for best expected performance.
    """
    def __init__(self):
        self.model = None
    
    def extract_features(self, Q):
        n = Q.shape[0]
        density = np.count_nonzero(Q) / (n*n)
        cond = np.linalg.cond(Q + np.eye(n)*1e-6)
        diag_ratio = np.abs(np.diag(Q)).sum() / np.abs(Q).sum()
        return np.array([n, density, cond, diag_ratio]).reshape(1, -1)
    
    def train(self, historical_data):  # historical_data: (features, best_solver_label)
        self.model = RandomForestClassifier()
        X = np.array([self.extract_features(Q) for Q, _ in historical_data])
        y = [label for _, label in historical_data]
        self.model.fit(X.reshape(len(X), -1), y)
    
    def recommend(self, Q):
        feats = self.extract_features(Q)
        return self.model.predict(feats)[0]
5. Data synthetic generator (data/synthetic_generator.py)
python
import numpy as np
import pandas as pd

def generate_reservoir_data(n_wells=10, seed=42):
    np.random.seed(seed)
    # Simulate realistic grid positions (x,y)
    x = np.random.uniform(0, 100, n_wells)
    y = np.random.uniform(0, 100, n_wells)
    distance = np.sqrt((x[:, None] - x[None, :])**2 + (y[:, None] - y[None, :])**2)
    
    porosity = np.random.beta(2, 5, n_wells) * 0.3   # 0..0.3
    permeability = np.random.lognormal(mean=1, sigma=0.5, size=n_wells) * 100  # mD
    pressure = np.random.uniform(2000, 4000, n_wells)  # psi
    
    # Save raw
    raw_df = pd.DataFrame({'x': x, 'y': y, 'porosity': porosity, 
                           'permeability': permeability, 'pressure': pressure})
    raw_df.to_csv('data/raw/reservoir_properties.csv', index=False)
    return porosity, permeability, pressure, distance
